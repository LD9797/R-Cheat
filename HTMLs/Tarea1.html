# 3. Eliminar outliers

# IQR
eliminate_outliers <- function(mydata) {
  quartiles <- quantile(mydata, probs=c(.25, .75), na.rm = FALSE)
  IQR <- IQR(mydata)
  Lower <- quartiles[1] - 1.5 * IQR
  Upper <- quartiles[2] + 1.5 * IQR
  clean_data <- subset(mydata, mydata > Lower & mydata < Upper)
  return (clean_data)
}
